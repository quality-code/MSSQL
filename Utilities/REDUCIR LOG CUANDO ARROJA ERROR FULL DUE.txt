--VERIFICA EL MODO DE RECUPERACION DE LA BD, SI ES SIMPLE NO PERMITIRA HACER BACKUP LOG
USE [COMPUFACIL] --AQUI EL NOMBRE DE LA BD
GO
SELECT name BD, recovery_model_desc [MODO DE RECUPERACION] 
   FROM sys.databases  
      WHERE name = 'COMPUFACIL'--AQUI EL NOMBRE DE LA BD
GO 
--************************************************************************
USE [COMPUFACIL] --AQUI EL NOMBRE DE LA BD
GO
BACKUP LOG [COMPUFACIL] --AQUI EL NOMBRE DE LA BD
   TO  DISK = N'C:\Program Files\Microsoft SQL Server\MSSQL14.SQLEXPRESS\MSSQL\Backup\COMPUFACIL' --RUTA DONDE SE GUARDARA EL BACKUP
DBCC SHRINKFILE (N'COMPUFACIL_log' , 0, TRUNCATEONLY)--NOMBRE DEL ARCHIVO QUE SE VA A REDUCIR
GO
