--VER USUARIOS CREADOS EN EL MES DEL INFORME
--REEMPLAZAR EL RANGO DE FECHAS SI ES NECESARIO
USE MSDB
GO
SELECT        ObjectName AS [USUARIO CREADO], ObjectType AS OBJETO, EventDate AS FECHA, LoginName AS [USUARIO CREO]
FROM            DatabaseChangeLogDbar
WHERE        (SqlCommand LIKE '%create%') AND (EventType = 'CREATE_LOGIN') AND (EventDate BETWEEN CONVERT(DATETIME, '2018-09-01 00:00:00', 102) AND CONVERT(DATETIME, '2018-09-30 00:00:00', 102)) OR
                         (EventType = 'CREATE_USER')
GO
